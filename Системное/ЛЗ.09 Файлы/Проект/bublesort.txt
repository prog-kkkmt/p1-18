//Сортировка пузырьком
#include <stdio.h>
#define MAX 100

int main() {
int n;
int mas[MAX];
scanf("%d", &n);

for (int i = 0; i != n; ++i){
	scanf("%d", &mas[i]);
}

for(int i = 0 ; i < n - 2; i++) { 
       // сравниваем два соседних элемента.
       for(int j = 0 ; j < n - i - 2 ; j++) {  
           if(mas[j] > mas[j+1]) {           
              // если они идут в неправильном порядке, то  
              //  меняем их местами. 
              int tmp = mas[j];
              mas[j] = mas[j+1] ;
              mas[j+1] = tmp; 
              }
       }
}

for (int i = 0; i != n; ++i){
	printf("%d ", mas[i]);
}
printf("\n");
return 0 ;
}
Шаги алгоритма:
1) Проходим по массиву, сравнивая j-ый элемент с j+1
2) Если j-ый элемент больше j+1, то меняем их местами
